/**
 * Addition function
 * @param {Number} summand_1 first summand
 * @param {Number} summand_2 second summand
 * @return {Number} result after addding summand_1 to summand_2
 */
const add = function (summand_1, summand_2) {
  let sum = summand_1 + summand_2;
  return sum
};
